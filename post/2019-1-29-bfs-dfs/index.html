深度优先遍历和广度优先遍历

· Invalid date · # blog

{{page.title}}

我们用二叉树作为数据结构，来写出深度优先遍历和广度优先遍历。

深度优先遍历

该算法先将当前结点的孩子全部遍历结束，再遍历同一级的节点

    // 递归
    let res = [];
    let dfs = function(node) {
        if (node) {
            res.push(node.val);
            dfs(node.left);
            dfs(node.right);
        }
    }
    // 非递归
    let dfs = function(nodes) {
        let res = [];
        let stack = [];
        stack.push(nodes);
        while (stack.length) {
            let node = stack.pop();
            res.push(node.val);
            node.right && stack.push(node.right); // 先压入右子树
            node.left && stack.push(node.left);
        }
        return res;
    }

广度优先遍历

当使用广度优先遍历的时候，先依次遍历兄弟节点，然后便利兄弟节点下的子节点，也就是按层次的去遍历。依次遍历根节点，然后是左孩子和右孩子。

    // 递归
    let res = [],
        stack = [],
        count = 0;

    stack.push(nodes);
    let bfs = function() {
        let node = stack[count];
        if (node) {
            res.push(node.val);
            stack.push(node.left);
            stack.push(node.right);
            count ++;
        }
        bfs();
    }

    // 非递归
    let bfs = function() {
        let res = [],
            stack = [],
            count = 0;

        stack.push(nodes);
        while(stack.length){
            let node = stack.shift();
            res.push(node.value);
            node.left && stack.push(node.left);
            node.right && stack.push(node.right);
        }
        return res;
    }
